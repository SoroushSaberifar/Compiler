abstract class Animal {
    protected String name;
    private int age;

    public Animal(String n, int a) {
        name = n;
        age = a;
    }

    public void display() {
        print("Name: ");
        print(name);
        print(", Age: ");
        print(age);
    }

    public abstract void makeSound(); 
}

class Dog extends Animal {
    private String breed;
    public int MAX_AGE;

    public Dog(String n, int a, String b) {
        name = n;
        age = a;
        breed = b;
        MAX_AGE = 20;
    }

    @Override
    public void makeSound() {
        print("Woof!");
    }

    public void play() {
        print("Dog is playing");
    }

    public void bark(int times) {
        int i;
        i = 0;
        while (i < times) {
            print("bark ");
            i = i + 1;
        }
    }

    public void forExample() {
        int sum;
        int j;
        sum = 0;
        for (j = 0; j < 10; j = j + 1) {
            sum = sum + j;
        }
        print(sum);
    }

    public int compareTo(Dog other) {
        int result;
        if (age > other.getAge()) {
            return 1;
        } else {
            if (age < other.getAge()) {
                return -1;
            } else {
                return 0;
            }
        }
        return result;
    }
    
    public int getAge() {
        return age;
    }
}
