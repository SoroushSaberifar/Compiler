class Calculator {

    private int result;
    public int counter;
    protected int limit;
    internal int flagCount;

    Calculator(int start) {
        result = start;
        counter = 0;
    }

    public int add(int a, int b) {
        int sum;
        sum = a + b;
        result = sum;
        return sum;
    }

    protected int multiply(int a, int b) {
        int temp;
        temp = 0;
        int i;
        i = 0;
        while (i < b) {
            temp = temp + a;
            i = i + 1;
        }
        return temp;
    }

    public int getResult() {
        return result;
    }

    public int twice() {
        int d;
        d = this.getResult() + this.getResult();
        return d;
    }
}

abstract class Shape {
    protected int side;

    public void setSide(int s) {
        side = s;
    }

    public abstract int area(int scale);
}

class Square extends Shape {

    @Override
    public int area(int scale) {
        int a;
        a = side * side * scale;
        return a;
    }
}
