class MainApp {
    public static void main(String[] args) {
        int x = 10;
        Calculator calc = new Calculator();
        int result = calc.compute(x, 5);
        print(result);
    }
}

abstract class BaseOp {
    protected int baseValue;
    
    public int getBase() {
        return baseValue;
    }
}

class Calculator implements Printable {
    private boolean isActive;
    private int baseValue;

    public Calculator() {
        baseValue = 100;
        isActive = true;
    }

    public int compute(int a, int b) {
        int sum = a + b;
        if (isActive) {
            sum = sum * 2;
        } else {
            sum = sum - 1;
        }
        return sum + baseValue;
    }

    @Override
    public String format(int val) {
        return "Result is printed";
    }
}

interface Printable {
    String format(int value);
}
